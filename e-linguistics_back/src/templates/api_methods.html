/nltk/tokenize

method: POST
example body: {
"input_text": "some sentence that needs tokenizing."
}
description: splits sentence into individual words and returns them as a string array
example response: ['some', 'sentence', 'that', 'needs', 'tokenizing', '.']

/nltk/find/<lang>/synsets

method: POST
example body: {
"search_word": "word"
}
description: searches for a word inside the wordnet with language code <lang> and returns array of info for each synset found
example response: [
{
"synset_offset": "synset offset and type combination, used as id"
"wn_synset": "synset name"
"definition": "synset definition"
}, ...
]

/cltk/lemmatize

method: POST
example body: {
"input_text": "word"
}
description: returns the initial form of the word (or words in a sentence), that can be found in a dictionary
example response: lemma

/db/inscription?phID=phid

method: GET
description: returns an inscription record based on the given phID
example response: Inscription object

/db/inscription

method: POST
example body: {
"inscription_text" : "...",
"phID" : "PH190736",
"dateFrom" : "-340",
"dateTo" : "-340",
"id_in_publication" : "...",
"general_type" : "αναθηματική",
"provenance" : "...",
"bibliography" : "....",
"words" : ["w1_synsetId", "w2_synsetId", ...]
}
description: inserts a new inscription in the database
example response: ObjectID
